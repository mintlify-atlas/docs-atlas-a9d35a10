## Overview

The Tooltip component wraps any element and displays additional information when the user hovers over or focuses on it.

## Basic Usage

```tsx
import Tooltip from '@newtonschool/grauity';
import Button from '@newtonschool/grauity';

function MyComponent() {
  return (
    <Tooltip content="This is a helpful tooltip">
      <Button variant="primary">Hover over me!</Button>
    </Tooltip>
  );
}
```

## Props

<ParamField path="content" type="React.ReactNode">
  The content to display inside the tooltip.
</ParamField>

<ParamField path="placement" type="string" default="top">
  Position of the tooltip relative to the wrapped element.
  
  **Available choices:** `top`, `bottom`, `left`, `right`, `top-start`, `top-end`, `bottom-start`, `bottom-end`, `left-start`, `left-end`, `right-start`, `right-end`
</ParamField>

<ParamField path="fixedPositioning" type="boolean" default={false}>
  Use fixed positioning instead of absolute.
</ParamField>

<ParamField path="hidden" type="boolean" default={false}>
  Hide the tooltip.
</ParamField>

<ParamField path="hideArrow" type="boolean" default={false}>
  Hide the tooltip arrow indicator.
</ParamField>

<ParamField path="defaultOpen" type="boolean" default={false}>
  Show the tooltip by default on mount.
</ParamField>

<ParamField path="isOpen" type="boolean | null" default={null}>
  Controlled state for tooltip visibility. When provided, the tooltip becomes controlled.
</ParamField>

<ParamField path="color" type="string" default="var(--text-emphasis-invert-primary-default, #ffffff)">
  Text color of the tooltip.
</ParamField>

<ParamField path="backgroundColor" type="string" default="var(--bg-subtle-invert-primary-default, #0b0c0e)">
  Background color of the tooltip.
</ParamField>

<ParamField path="config" type="TooltipConfig">
  Configuration object for customizing tooltip appearance.
  
  **Structure:** `{ tooltip: { maxWidth?: string, padding?: string, backgroundColor?: string } }`
</ParamField>

<ParamField path="recomputedTrigger" type="any">
  A value that triggers tooltip position recalculation when changed.
</ParamField>

<ParamField path="children" type="React.ReactNode" required>
  The element that the tooltip wraps.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS class name.
</ParamField>

```tsx

  content="Tooltip without arrow"
```

## Controlled Tooltip

```tsx
import { useState } from 'react';

function ControlledTooltipExample() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div>
      <Tooltip content="Controlled tooltip" isOpen={isOpen}>
        <Button>Hover me</Button>
      </Tooltip>
      
      <Button onClick={() => setIsOpen(!isOpen)}>
        Toggle Tooltip
      </Button>
    </div>
  );
}
```
